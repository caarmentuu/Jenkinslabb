package se.iths;

import java.time.LocalDate;

public class RunningSession {
    private String identifier;
    private double distance; // in km
    private Time time;
    private LocalDate date;

    public RunningSession() {
        setDateNow();
    }

    public RunningSession(double distance, Time time) {
        this.distance = distance;
        this.time = time;
        setDateNow();
    }

    public RunningSession(double distance, Time time, String date) {
        this.distance = distance;
        this.time = time;
        this.date = LocalDate.parse(date);
    }

    private void setDateNow() {
        this.date = LocalDate.now();
    }

    public double getDistance() {
        return distance;
    }

    public Time getTime() {
        return time;
    }

    public String getIdentifier() {
        return identifier;
    }

    public void setIdentifier(String identifier) {
        this.identifier = identifier;
    }

    public double getAverageSpeed() {
        return distance / time.getTotalHours();
    }

    public double getKilometerTime() {
        return time.getTotalMinutes() / distance;
    }

    @Override
    public String toString() {
        return "RunningSession{" +
                "identifier='" + identifier + '\'' +
                ", distance=" + distance +
                ", time=" + time +
                ", date=" + date +
                '}';
    }
}
